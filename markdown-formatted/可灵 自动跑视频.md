```python
import os
import pandas as pd
from selenium import webdriver
from selenium.webdriver.chrome.service import Service
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.common.by import By
from selenium.webdriver.common.action_chains import ActionChains
import time
import chardet
import logging

# 设置日志
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')

# 设置 ChromeDriver 路径
driver_path = r"C:\Program Files\Google\Chrome\chromedriver-win64\chromedriver.exe"  # 替换为实际的 ChromeDriver 路径
service = Service(driver_path)
driver = webdriver.Chrome(service=service)

# 输入图片文件夹路径和 CSV 文件路径
image_folder = r"C:\Users\Admin\Desktop\test"  # 替换为实际的图片文件夹路径
csv_file = r"C:\Users\Admin\Desktop\aaa.csv"  # 替换为实际的 CSV 文件路径

# 检测文件编码
with open(csv_file, 'rb') as f:
    result = chardet.detect(f.read())
    encoding = result['encoding']
    logging.info(f"检测到的文件编码: {encoding}")

# 使用检测到的编码读取 CSV 文件
tasks = pd.read_csv(csv_file, encoding=encoding)

# 打开网页
driver.get('https://klingai.kuaishou.com/text-to-video/new')

# 创建一个显式等待对象
wait = WebDriverWait(driver, 30)

for index, row in tasks.iterrows():
    image_id = row['序号']  # 获取序号
    text_content = row['Prompt 中文']  # 获取文本内容

    # 根据序号找到图片路径
    image_path_jpg = os.path.join(image_folder, f"{image_id}.jpg")
    image_path_png = os.path.join(image_folder, f"{image_id}.png")

    # 确定图片是否存在
    if os.path.exists(image_path_jpg):
        image_path = image_path_jpg
    elif os.path.exists(image_path_png):
        image_path = image_path_png
    else:
        logging.warning(f"未找到序号为 {image_id} 的图片，跳过该项，查找路径: {image_path_jpg}, {image_path_png}")
        continue

    # 等待页面加载完成
    try:
        wait.until(EC.presence_of_element_located((By.CSS_SELECTOR, "div.el-textarea.prompt textarea")))
        logging.info("页面加载成功。")
    except Exception as e:
        logging.error(f"页面未能正确加载: {str(e)}")
        continue

    # 清除已上传的图片（添加的步骤）
    try:
        delete_image_button = wait.until(EC.element_to_be_clickable((By.CSS_SELECTOR, "div.delete-btn.clickable")))
        delete_image_button.click()
        logging.info("成功删除已上传的图片。")
        time.sleep(2)  # 等待图片删除完成
    except Exception as e:
        logging.warning(f"未找到删除图片按钮，可能没有已上传的图片: {str(e)}")

    # 清空上次的文本
    try:
        textarea = wait.until(EC.presence_of_element_located((By.CSS_SELECTOR, "div.el-textarea.prompt textarea")))
        textarea.clear()
        logging.info("成功清空上一次的文本历史。")
    except Exception as e:
        logging.error(f"无法清空文本历史，错误: {str(e)}")
        continue

    # 步骤 1: 切换到“图生视频”选项卡
    try:
        video_tab = wait.until(EC.element_to_be_clickable((By.XPATH, "//span[text()='图生视频']")))
        video_tab.click()
        logging.info("成功切换到图生视频。")
    except Exception as e:
        logging.error(f"无法切换到图生视频，错误: {str(e)}")
        continue

    # 步骤 2: 上传图片
    try:
        upload_input = wait.until(EC.presence_of_element_located((By.CSS_SELECTOR, "input.el-upload__input")))
        upload_input.send_keys(image_path)
        logging.info(f"成功上传图片: {image_path}")
        time.sleep(5)  # 等待图片上传完成
    except Exception as e:
        logging.error(f"无法上传图片，错误: {str(e)}")
        continue

    # 步骤 3: 输入文本（如果有文本）
    if pd.notna(text_content) and text_content.strip():
        try:
            textarea = wait.until(EC.presence_of_element_located((By.CSS_SELECTOR, "div.el-textarea.prompt textarea")))
            textarea.send_keys(text_content)
            logging.info(f"成功输入文本: {text_content}")
        except Exception as e:
            logging.error(f"无法输入文本，错误: {str(e)}")
            continue
    else:
        logging.info("没有文本输入，跳过文本输入步骤。")

    # 步骤 4: 点击“高性能”选项
    try:
        performance_option = wait.until(EC.element_to_be_clickable((By.XPATH, "//span[text()='高性能']")))
        performance_option.click()
        time.sleep(3)
        logging.info("成功选择高性能选项。")
    except Exception as e:
        logging.error(f"无法选择高性能选项，错误: {str(e)}")
        continue

    # 步骤 5: 点击“5s”选项
    try:
        duration_option = wait.until(EC.element_to_be_clickable((By.XPATH, "//span[text()='5s']")))
        duration_option.click()
        time.sleep(3)
        logging.info("成功选择5秒选项。")
    except Exception as e:
        logging.error(f"无法选择5秒选项，错误: {str(e)}")
        continue

    # 步骤 7: 点击生成按钮
    try:
        final_button = wait.until(EC.element_to_be_clickable(
            (By.XPATH, "//button[contains(@class, 'generic-button') and contains(@class, 'primary')]")))
        driver.execute_script("arguments[0].scrollIntoView(true);", final_button)
        final_button.click()
        logging.info(f"已提交生成请求，序号: {image_id}")
        time.sleep(10)  # 等待生成过程完成
    except Exception as e:
        logging.error(f"无法提交生成请求，错误: {str(e)}")
        continue

logging.info("批量操作完成，关闭浏览器。")
driver.quit()
```
